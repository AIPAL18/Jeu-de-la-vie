430 def event(self, even: QEvent) -> bool:
431 """
432 Réimplémentation de event hérité de QMainWindow
433 Entrées:
434 self: JeuDeLaVieGUI
435 even: QEvent (et les classes qui en hérite)
436 Sortie:
437 bool
438 Rôle:
439 Capture les événements de QMainWindow, les traites, puis les rends.
440 """
441 # Si le type de l'évènement est StatusTipEvent
442 if even.type() is QEvent.Type.StatusTip:
443 # On précise le type de l'évènement pour être sûr de pouvoir
444 # accéder aux méthodes (optionnel)
445 even: QStatusTipEvent = even
446 # Définit le texte de bar_info sur le conseil de l'évènement
447 self.bar_info.setText(even.tip())
448 """
449 Fonctionnement:
450 Chaque objet possédant l'attribut statusTip émet un signal
451 QStatusTipEvent lorsque la souris le survole. Ce signal comporte un
452 tip, un conseil, concernant le dit objet. Il est par défaut
453 réceptionné par statusBar de QMainWindow, mais il peut être
454 intercepté et affiché autre part.
455 """
456
457 # On rend l'évènement
458 return super().event(even)
