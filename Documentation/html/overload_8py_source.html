61def get_signature_complete(obj, sig: tuple = ()) -> tuple[str]:
62 """
63 Entrées:
64 obj: Any
65 obj est une variable quelconque
66 sig: tuple
67 signature de obj jusqu'alors
68 Sortie:
69 tuple[str]
70 Rôle:
71 Retourne l'arbre de signature de obj
72 Exemple:
73 >>> a: int = 1
74 >>> get_inherited_signature(a)
75 ['int', 'object']
76 Explications:
77 Le but de cette fonction est de savoir de quelleS classeS hérite obj.
78 Plus obj est abstrait, plus sont arbre de signature sera grand.
79 L'arbre ne possède qu'une dimension pour facilité son traitement.
80 Addendum:
81 Cette fonction est la raison pour laquelle j'ai recodé ce module.
82 Ce dernier ne permet pas de faire de surcharge 'intelligente'.
83 Il est évident les parmatères suivants de type:
84 list, int, int
85 correspond à la signature suivante:
86 list, int, object
87 car int hérite de object. Or l'héritage doit respecter le principe de
88 substitution de Liskov.
89 """
90 # Si c'est le premier appelle
91 if len(sig) == 0:
92 # On redéfinit obj par sa classe
93 obj = obj.__class__
94 # on attribut le singleton du nom de la classe d'obj à sig
95 sig = (obj.__name__,)
96 # On retourne get_inherited_signature(obj, sig)
97 return get_signature_complete(obj, sig)
98 # Si obj est object
99 elif obj.__name__ == object.__name__:
100 # On retourne la signature complète plus object
101 return sig + (obj.__name__,)
102 # Sinon
103 else:
104 # parent_sig est initialisé avec les classes directement parentes d'obj
105 parent_sig = obj.__bases__
106 # Si obj à plus d'un parent directe
107 if type(parent_sig) is tuple:
108 # Pour chaque parent
109 for par in parent_sig:
110 # Si le parent est object
111 if par.__name__ == object.__name__:
112 # on ajoute ('object',) à sig
113 sig += (par.__name__,)
114 # Sinon
115 else:
116 # on ajoute l'arbre de signature du parent
117 sig += get_signature_complete(par, (par.__name__,))
118 # On retourne l'arbre de signature de obj
119 return sig
120 # Si sig n'a qu'un parent directe
121 else:
122 # On retourne sig plus son type plus son arbre de signature
123 return sig + (obj.__name__,) + (
124 get_signature_complete(parent_sig, sig),)

127class Overload(object):
128 """
129 Hérite de:
130 object
131 Rôle:
132 Permet de surcharger une (uniquement) méthode (uniquement) et met en
133 oeuvre le principe de substitution de Liskov.
134 Exemple:
135 >>> @Overload # (il est préférable de l'utiliser en décorateur)
136 >>> @signature(int) # (obligatoire !!!)
137 >>> def double(valeur: int) -> int:
138 >>> return int(valeur * 2)
139 >>> @double.overload # Pas de majuscule à overload
140 >>> @signature(float) # (obligatoire !!!)
141 >>> def double(valeur: float) -> float:
142 >>> return float(valeur * 2)
143 >>> double(.2) # .2 est de type float
144 0.4 # float
145 >>> double(5) # 5 est de type int
146 10 # int
147 Addendum:
148 Overload est dérivée, comme signature, de pythonlangutil.overload donc
149 les variables sont en englais.
150 """
151 def __init__(self, func: callable) -> None:
152 """
153 Entrées:
154 self: Overload
155 func: callable (fonction à surcharger)
156 Sortie:
157 None (ctor)
158 Rôle:
159 Construit un nouvel objet Overload
160 """
161 # Déclaration de l'attribut owner pour la méhtode magique __get__
162 self.owner = None
163 # Déclaration d'une liste de tuple où les tuples sont les signatures
164 # des méthodes qui se surchargent
165 self.signatures: list[tuple] = []
166 # Liste des méthodes qui se surchargent
167 self.methods: list[callable] = []
168 # On ajoute la méthode surchargée à methods
169 self.methods.append(func)
170 # On ajoute la signature de la méthode à signatures
171 self.signatures.append(func.signature)

192 def __call__(self, *args, **kwargs):
193 """
194 Entrée:
195 self: Overload
196 args: Any (arguments passés à la fonction surchargée)
197 kwargs: Any (couples (clé;arg) passés à la fonction surchargée)
198 Sortie:
199 Résultat de la fonction surchargé qui correspond à la signature des
200 arguments passés en paramètre.
201 Rôle:
202 Trouve la fonction correspondant à la signature des arguments
203 passés en paramètre, l'execute et retourne sa valeur de retour.
204 """
205 # Signature des arguments
206 signature = []
207 # Pour chaque argument sans clé
208 for arg in args:
209 # On ajoute le type des arguments sans clé
210 signature.append(arg.__class__.__name__)
211 # Pour chaque clé (_), argument (v)
212 for _, v in kwargs:
213 # On ajoute le type des arguments qui ont une clé
214 signature.append(v.__class__.__name__)
215 # Signature doit être un tuple car les signatures sont stockée sous
216 # forme de tuple
217 signature = tuple(signature)
218 # si la signature exacte est enregistrée
219 if signature in self.signatures:
220 # On trouve l'indice correspondant à cet signature
221 index = self.signatures.index(signature)
222 # si la signature exacte est n'est pas enregistrée
223 else:
224 # extrait l'arbre d'héritage pour chaque argument de la signature
225 inherited_signature = []
226 # Pour chaque argument sans clé
227 for arg in args:
228 # On ajoute l'arbre d'héritage des arguments sans clé
229 inherited_signature.append(get_signature_complete(arg))
230 # Pour chaque clé (_), argument (v)
231 for _, v in kwargs:
232 # On ajoute l'arbre d'héritage des arguments qui ont une clé
233 signature.append(get_signature_complete(v))
234
235 # signature éligibles
236 sig_eligibles = []
237 # Pour chaque signature des signatures enregistrées
238 for sig in self.signatures:
239 # Si son nombre d'argument correspond à la signature des args
240 if len(inherited_signature) == len(sig):
241 # On l'ajoute aux signature éligibles
242 sig_eligibles.append(sig)
243
244 # Index négatif -> erreur
245 index = -1
246 # Variable d'arrêt, False s'il n'y a pas de signature éligible
247 existe = len(sig_eligibles) > 0 # bool
248 # Itérateur i
249 i = 0
250 # Pour chaque signature
251 while i < len(sig_eligibles) and existe:
252 # Itérateur j
253 j = 0
254 # Variable d'arrêt de la seconde boucle
255 correspond = True
256 # Pour chaque type de la signature
257 while j < len(sig_eligibles[i]) and correspond:
258 # Si le type n'est pas dans les type hérité du paramètre
259 # c'est que ce n'est pas le bon type
260 if sig_eligibles[i][j] not in inherited_signature[j]:
261 # On stop la boucle
262 correspond = False
263 # Incrémentation de l'itérateur
264 j += 1
265 # Si la signature correspond
266 if correspond:
267 # On recherche l'indice car la signature est unique
268 # (cf self.overload)
269 index = self.signatures.index(sig_eligibles[i])
270 # Incrémentation de l'itérateur
271 i += 1
272 # Si la signature ne correspond à aucune des méthodes enregistrées
273 if not existe or index < 0:
274 # On lève une exception
275 raise Exception("There is no overload for this method with " +
276 "this signature.")
277
278 # On execute la méthode avec ses arguments et on retourne sa valeur de
279 # retour
280 return self.methods[index](self.owner, *args, **kwargs)

282 def overload(self, func: callable):
283 """
284 Entrée:
285 self: Overload
286 func: callable
287 Sortie:
288 Overload
289 Rôle:
290 Ajoute une nouvelle surcharge
291 Explications:
292 Faire:
293 >>> @Overload
294 >>> @signature(int)
295 >>> def double(valeur: int) -> int:
296 >>> return int(valeur * 2)
297 C'est faire:
298 >>> def double(valeur: int) -> int:
299 >>> return int(valeur * 2)
300 >>> double = signature('int', double)
301 >>> double = Overload(double)
302
303 Donc quand on fait
304 >>> @double.overload # appelle la méthode overload
305 >>> @signature(float)
306 >>> def double2(valeur: float) -> float:
307 >>> return float(valeur * 2)
308 On fait:
309 >>> def double2(valeur: float) -> float:
310 >>> return float(valeur * 2)
311 >>> double2 = signature(double2)
312 >>> double2 = double.overload(double2)# appelle la méthode overload
313 En pratique, il ne faut pas changer le nom de la méthode quand on
314 la surcharge.
315
316 https://www.geeksforgeeks.org/decorators-in-python/
317 """
318 # Si la signature est unique
319 if func.signature not in self.signatures:
320 # On ajoute la méthode à methods
321 self.methods.append(func)
322 # On ajoute la signature de func à signatures
323 self.signatures.append(func.signature)
324 # Retourne l'objet Overload
325 return self
326 # Si la signature n'est pas unique
327 else:
328 # On lève une exception
329 raise Exception("There is no overload for this method with this" +
330 " signature: the signature is already taken.")
