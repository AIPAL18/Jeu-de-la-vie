195 # Abscisse de côté droit de la scène
196 droit = self.sceneRect().right()

200 if not self.est_souris_dans_scene:

208 i = int((y - haut) // 50)

210 j = int((x - gauche) // 50)

212 if event.buttons() is Qt.MouseButton.LeftButton:
213 # On définit l'état de la cellule touchée par la souris sur
214 # Vivant

216 # Si le bouton droit de la souris est pressé

218 # On définit l'état de la cellule touchée par la souris sur

322 """

342 # Récupère le voisin du haut gauche
343 hg = int(self.valeur_case(ord - 1, absc - 1))
344 # Récupère le voisin de droite
345 d = int(self.valeur_case(ord, absc + 1))
346 # Récupère le voisin de gauche
347 g = int(self.valeur_case(ord, absc - 1))
